CREATE TEMP TABLE $region_RESERVATIONS_TEMP AS
(SELECT
project_id,
project_number,
reservation_name,
ignore_idle_slots,
slot_capacity,
ddl,
edition,
target_job_concurrency,
autoscale.current_slots,
autoscale.max_slots
FROM `admin_project_id.region-$region`.INFORMATION_SCHEMA.RESERVATIONS);



EXPORT DATA OPTIONS(
  uri='gs://example-bucket/folder_name/$region_RESERVATIONS_*.csv',
  format='CSV',
  overwrite=true,
  header=true,
  field_delimiter = ',') AS
   SELECT * from $region_RESERVATIONS_TEMP;
